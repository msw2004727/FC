### P1-⑥ approve 流程原子化(1-2 小時)

**檔案**:[tournament-friendly-detail.js:186-208](js/modules/tournament/tournament-friendly-detail.js:186)、[firebase-crud.js](js/firebase-crud.js)

**修法**:新增 `reviewTournamentApplicationAtomic(tournamentId, applicationId, action, reviewer)`:
```javascript
async reviewTournamentApplicationAtomic(tournamentId, applicationId, action, reviewer) {
  const tournamentRef = await this._getTournamentDocRefById(tournamentId);
  const applicationRef = tournamentRef.collection('applications').doc(applicationId);
  
  return db.runTransaction(async (tx) => {
    const [tournamentSnap, appSnap] = await Promise.all([
      tx.get(tournamentRef),
      tx.get(applicationRef),
    ]);
    if (!appSnap.exists) throw new Error('APPLICATION_NOT_FOUND');
    const application = appSnap.data();
    if (application.status !== 'pending') throw new Error('APPLICATION_NOT_PENDING');
    
    if (action === 'approve') {
      // 在 transaction 內查 entries count → 名額 atomic
      const entriesSnap = await tx.get(tournamentRef.collection('entries'));
      const approvedCount = entriesSnap.docs.filter(d => {
        const s = d.data().entryStatus;
        return s === 'host' || s === 'approved';
      }).length;
      const teamLimit = tournamentSnap.data()?.friendlyConfig?.teamLimit || 4;
      if (approvedCount >= teamLimit) throw new Error('TOURNAMENT_FULL');
      
      // 寫 entry + 寫 app + 同步 registeredTeams summary 一次完成
      tx.set(tournamentRef.collection('entries').doc(application.teamId), {
        teamId: application.teamId, /* ... */
        entryStatus: 'approved',
        approvedAt: firebase.firestore.FieldValue.serverTimestamp(),
        approvedByUid: reviewer.uid,
      }, { merge: true });
      
      const newRegisteredTeams = [
        ...entriesSnap.docs.map(d => d.id),
        application.teamId,
      ];
      tx.update(tournamentRef, { 
        registeredTeams: newRegisteredTeams,
        updatedAt: firebase.firestore.FieldValue.serverTimestamp(),
      });
    }
    
    tx.update(applicationRef, {
      status: action === 'approve' ? 'approved' : 'rejected',
      reviewedAt: firebase.firestore.FieldValue.serverTimestamp(),
      reviewedByUid: reviewer.uid,
      reviewedByName: reviewer.displayName,
      updatedAt: firebase.firestore.FieldValue.serverTimestamp(),
    });
  });
}
```

match /applications/{applicationId} {
  function isValidTournamentApplicationCreate() {
    let data = request.resource.data;
    return data.status == 'pending'
      && data.requestedByUid == request.auth.uid
      && data.teamId is string
      && applicationId == 'ta_' + data.teamId
      && data.keys().hasOnly(['id', 'teamId', 'teamName', 'teamImage', 'status', 
                                'requestedByUid', 'requestedByName', 'appliedAt', 
                                'messageGroupId', 'createdAt', 'updatedAt']);
  }
  
  function isValidTournamentApplicationUpdate() {
    let data = request.resource.data;
    let prev = resource.data;
    return data.teamId == prev.teamId
      && data.requestedByUid == prev.requestedByUid
      && data.keys().hasOnly([...prev.keys(), 'status', 'reviewedAt', 'reviewedByUid', 
                                'reviewedByName', 'updatedAt']);
  }
  
  allow read: if canManageTournamentScope()
    || ((resource.data.teamId is string) && isCurrentUserInTeam(resource.data.teamId));
  allow create: if tournamentExists()
    && isAuth()
    && hasString('teamId')
    && isCurrentUserTeamCaptainOrLeader(request.resource.data.teamId)
    && isValidTournamentApplicationCreate();
  allow update: if tournamentExists() 
    && canManageTournamentScope()
    && isValidTournamentApplicationUpdate();
  allow delete: if tournamentExists() && canManageTournamentScope();
}
